 57:     /**
 58:      * Konstruktor. Der Konstruktor kann auf drei Arten aufgerufen werden:
 59:      * 
 60:      * 1. Ohne Übergabe von Parametern: `new NamiWrapper()`. Login-Daten müssen dann
 61:      *    der ´login´-Funktion übergeben werden.
 62:      * 
 63:      * 2. Mit Übergabe von Nutzername und Passwort: `new NamiWrapper('123456', 'password1234')`.
 64:      *    Ein Login kann mittels der `login`-Funktion durchgeführt werden oder er
 65:      *    wird automatisch bei Bedarf durchgeführt.
 66:      * 
 67:      * 3. Mit Übergabe eines Arrays: `new NamiWrapper(['username' => '123456',
 68:      *    'password' => 'password1234', 'iniFile' => 'custom.ini'])`. Folgende Schlüssel
 69:      *    können gesetzt werden:
 70:      * 
 71:      *    * iniFile: Standardwert: `nami.ini`. Eine Konfigurationsdatei, die im
 72:      *      wesentlichen die selben Parameter  erwartet, wie $config. Neben `nami.ini`
 73:      *      wird auch `namitest.ini` mitgeliefert. Eigene Konfigurationsdateien
 74:      *      können selbstverständlich auch erstellt werden. Soll keine Konfigurationsdatei
 75:      *      verarbeitet werden, so kann `no.ini` übergeben werden.
 76:      * 
 77:      *    * serverURL: Standardwert: `https://nami.dpsg.de`. URL des Servers, auf
 78:      *      dem die NaMi-API verfügbar ist.
 79:      * 
 80:      *    * serverApiPath: Standardwert: `/ica/rest/`. Pfad unter dem die API erreicht
 81:      *      werden kann.
 82:      * 
 83:      *    * serverLoginLoginValue: Standartwert: `API`. Wert, der bei Login-Anfragen
 84:      *      an die API im Feld `Login` geschickt wird.
 85:      * 
 86:      *    * APIVersionMajor: Standardwert: `1`. Hauptversionsnummer der API.
 87:      * 
 88:      *    * APIVersionMinor: Standardwert: `2`. Nebenversionsnummer der API.
 89:      * 
 90:      *    * timeout: Standartwert: `7.0`. Zeit, nach der Anfragen an die API abgebrochen
 91:      *      werden.
 92:      * 
 93:      *    * saveCookieInFile: Standardwert: `false`. Legt fest, ob die aktuelle
 94:      *      Session in einer Datei gespeichert werden soll.
 95:      * 
 96:      *    * cookieFile: Standardwert: `tmp/cookie.tmp`. Pfad zu der Datei, in der
 97:      *      ggf. die aktuelle Session gespeichert werden soll. Achtung! Stellen
 98:      *      Sie sicher, dass auf diese Datei nicht aus dem Weg zugegriffen werden
 99:      *      kann. Zur Sicherheit werden nur 1. absolute Pfade und 2. Pfade der Form
100:      *      `pfad/zu/datei.ext` akzeptiert. Zweite werden an `pfad/zu/simpleNaMiAPI/`
101:      *      angehängt.
102:      * 
103:      *    * createDirectoryMode: Standardwert `0600`. Zugriffsberechtigung, mit
104:      *      der ggf. das Verzeichnis der Cookie-Datei erstellt wird (Suche nach
105:      *      chmod).
106:      * 
107:      *    * unsername: Standardwert: `null`. Benutzername, der beim Login verwendet
108:      *      werden soll. Kann auch bei einer expliziten Login-Anfrage erst übergeben
109:      *      werden (vgl. Funktion `login()`).
110:      * 
111:      *    * password: Standardwert: `null`. Passwort, das beim Login verwendet
112:      *      werden soll. Kann auch bei einer expliziten Login-Anfrage erst übergeben
113:      *      werden (vgl. Funktion `login()`).
114:      * 
115:      *    * apiSessionNameRef: Standardmäßig nicht gesetzt. Wert muss als Referenz
116:      *      übergeben werden! Referenziert immer den Namen der aktuellen Session.
117:      *      Kann genutzt werden, um die aktuelle Session in der PHP-Session des
118:      *      Nutzers zu speichern: `$namiOpts['apiSessionNameRef'] =& $_SESSION['apiSessionName'];
119:      *      (…) $nw = new NamiWrapper($namiOpts);`
120:      * 
121:      *    * apiSessionTokenRef: Standardmäßig nicht gesetzt. Wert muss als Referenz
122:      *      übergeben werden! Referenziert immer das Token der aktuellen Session.
123:      *      Kann genutzt werden, um die aktuelle Session in der PHP-Session des
124:      *      Nutzers zu speichern: `$namiOpts['apiSessionTokenRef'] =& $_SESSION['apiSessionToken'];
125:      *      (…) $nw = new NamiWrapper($namiOpts);`
126:      * 
127:      * @param mixed $config Der Benutzername, ein Array mit Parametern oder `null`.
128:      * @param string $password Das Passwort oder `null`.
129:      */
130:     public function __construct($config = null, $password = null) {
131:         // hard coded standard values;
132:         $this->config['iniFile'] = 'nami.ini';
133:         $this->config['serverURL'] = 'https://nami.dpsg.de';
134:         $this->config['serverApiPath'] = '/ica/rest/';
135:         $this->config['serverLoginLoginValue'] = 'API';
136:         $this->config['APIVersionMajor'] = 1;
137:         $this->config['APIVersionMinor'] = 2;
138:         $this->config['timeout'] = 7.0;
139:         $this->config['saveCookieInFile'] = false;
140:         $this->config['cookieFile'] = 'tmp/cookie.tmp';
141:         $this->config['createDirectoryMode'] = 0600;
142:         $this->config['username'] = null;
143:         $this->config['password'] = null;
144:         // check if given parameters are valid and throw exception if necessary
145:         if (!((null == $config && null == $password) || (is_string($config) && is_string($password)) || (is_array($config) && null == $password))) {
146:             if (is_string($password)) {
147:                 $password = '*****';
148:             }
149:             if (is_array($config) && isset($config['password'])) {
150:                 $config['password'] = '*****';
151:             }
152:             throw new InvalidArgumentException('The constructor of NamiWrapper was '
153:                     . 'provided with invalid parameters. Expected are either no '
154:                     . 'parameters, two strings or one array! Parameters that were '
155:                     . 'provided: ' . var_export($config, true) . ', ' . 
156:                     var_export($password, true) . '.');
157:         }
158:         // in case username and password are directly provided, convert to array
159:         if ($password) {
160:             $config = ['username' => $config, 'password' => $password];
161:         }
162:         // check if custom ini-file is provided
163:         if (isset($config['iniFile']) && is_string($config['iniFile'])) {
164:             $this->config['iniFile'] = $config['iniFile'];
165:         }
166:         // load ini file, if filename is not 'no.ini' and merge with $this->config
167:         $configIni = ('no.ini' != $this->config['iniFile']) ?  @parse_ini_file($this->config['iniFile'], false, INI_SCANNER_TYPED) : [];

169:             $this->config = array_merge($this->config, $configIni);
170:         } else { // throw exception if ini file could not be loaded
171:             throw new RuntimeException('ini file ' . var_export($this->config['iniFile'], true)
172:                     . ' could not be opened or parsed!');
173:         }
174:         // merge $config with $this->config
175:         $this->config = array_merge($this->config, $config);
176:         // process references to api sessions
177:         $this->apiSessionName =& $this->config['apiSessionNameRef'];
178:         $this->apiSessionToken =& $this->config['apiSessionTokenRef'];

182:             if(preg_match('/^[^\.\/][^:]/', $this->config['cookieFile'])) {

184:             }

187:                         . ' path for $config[\'cookieFile\']. You may also provide'
188:                         . ' a path in the form of \'path/to/file.ext\'. which will'
189:                         . ' be appended to the directory of simplenamiapi. You provided '
190:                         . var_export($this->config['cookieFile'], true) . '.');
191:             }
192:             $cookie = explode('=', @file_get_contents($this->config['cookieFile']));
193:             if (is_string($cookie[0]) && isset($cookie[1]) && is_string($cookie[1])) {
194:                 $this->apiSessionName = $cookie[0];
195:                 $this->apiSessionToken = $cookie[1];
196:             }
197:         }

222:      * @param mixed $arg1 Hauptversionsnummer (int) oder `login` (string) oder `null`.

255:     /**
256:      * Funktion zum Durchführen von Anfragen an die Nami.
257:      * @param string $method Die zu verwendende HTTP-Methode (siehe Klassenkonstanten).
258:      * @param string $resource Die betroffene Ressource (Teil der URL, der nach `service/`
259:      * kommt).
260:      * @param object $content `null` oder Objekt, abhängig von Methode und Ressource.
261:      * Wird JSON- bzw. URL-kodiert.
262:      * @param mixed $apiMajor Hauptversionsnummer der API (int) oder `login`, falls
263:      * eine Login-Abfrage geschickt werden soll. Optional, falls ein Standardwert gesetzt wurde.
264:      * @param mixed $apiMinor Nebenversionsnummer der API (int) oder `null`, falls
265:      * eine Login-Abfrage geschickt werden soll. Optional, falls ein Standardwert gesetzt wurde.
266:      * @param string $encoding Optional. `NamiWrapper::ENCODING_JSON` sorgt für eine
267:      * JSON-Kodierung (Standardwert für POST- und PUT-Anfragen), `NamiWrapper::ENCODING_URL`
268:      * sorg für URL-Kodierung (wird vmtl. ausschließlich bei Login-Anfragen erwartet),
269:      * `NamiWrapper::ENCODING_JSON_MANUAL` erwartet bereits JSON-kodierten Inhalt und
270:      * `NamiWrapper::ENCODING_NO_CONTENT` (Standard für GET- und DELETE-Anfragen) bedeutet,
271:      * dass kein Inhalt gesendet wird.
272:      * @param boolean $autoLogin Optional. Standardmäßig wird ein automatischer Login-Versuch
273:      * unternommen. Falls `$autoLogin` `false` ist, wird dies unterlassen.
274:      */
275:     public function request($method, $resource, $content = null, $apiMajor = null, $apiMinor = null, $encoding = null, $autoLogin = true) {
276:         // check if method is valid method
277:         if (!self::_isValidMethod($method)) {
278:             throw new InvalidArgumentException('request expects a valid HTTP method'
279:                     . ' string. Value provided: ' . var_export($method, true) . '.');
280:         }
281:         // check if encoding is valid encoding and populate with default value
282:         if (null == $encoding) {
283:             switch ($method) {
284:                 case self::METHOD_POST:
285:                 case self::METHOD_PUT:
286:                     $encoding = self::ENCODING_JSON;
287:                     break;
288:                 case self::METHOD_GET:
289:                 case self::METHOD_DELETE:
290:                     $encoding = self::ENCODING_NO_CONTENT;
291:                     break;
292:             }
293:         }
294:         if (!self::_isValidEncoding($encoding)) {
295:             throw new InvalidArgumentException('request excepts a valid encoding '
296:                     . 'string or null. Value provided: ' . var_export($encoding, true) . '.');
297:         }
298:         // Add basic HTTP properties
299:         $httpsOpts = ['http' => [
300:                 'method' => $method,
301:                 'header' => 'Cookie: ' . $this->apiSessionName . '=' . $this->apiSessionToken . "\r\n",
302:                 'timeout' => $this->config['timeout']
303:             ],
304:         ];
305:         // add content if available
306:         switch($encoding) {
307:             case self::ENCODING_JSON:
308:                 $contentEncoded = json_encode($content);
309:                 $httpsOpts['http']['header'] .= 'Content-type: ' . self::MIME_JSON . "\r\n";
310:                 break;
311:             case self::ENCODING_URL:
312:                 $contentEncoded = http_build_query($content);
313:                 $httpsOpts['http']['header'] .= 'Content-type: ' . self::MIME_URL_ENCODED . "\r\n";
314:                 break;
315:             case self::ENCODING_JSON_MANUAL:
316:                 $contentEncoded = $content;
317:                 $httpsOpts['http']['header'] .= 'Content-type: ' . self::MIME_JSON . "\r\n";
318:                 break;
319:         }
320:         if (self::ENCODING_NO_CONTENT != $encoding) {
321:             $httpsOpts['http']['content'] = $contentEncoded;
322:         }
323:         // create http-context
324:         $context = stream_context_create($httpsOpts);
325:         // build url
326:         try {
327:             $apiVersion = $this->_apiVersion2UrlString($apiMajor, $apiMinor);
328:         } catch(InvalidArgumentException $e) {
329:             throw new UnexpectedValueException('There is something wrong with the'
330:                     . ' API version numbers provided when calling method request,'
331:                     . ' provided constructing this NamiWrapper or stored in the used'
332:                     . ' ini file.', 0, $e);
333:         }
334:         $url = $this->config['serverURL'] . $this->config['serverApiPath'] . $apiVersion . $resource;
335:         // make HTTP request; disable warnings since failed to open stream is likely to occour
336:         $tempErr = error_reporting();
337:         error_reporting($tempErr & ~E_WARNING);
338:         $response = file_get_contents($url, false, $context);
339:         error_reporting($tempErr);
340:         // report error if necessary
341:         if (false == $response) {
342:             $result = new NWRequestAnswer('Could not acces API located at ' . var_export($url, true) . '.', self::STATUS_CODE_CONNECTION_FAILED);
343:         } else { // otherwise decode response
344:             $result = json_decode($response);
345:         }
346:         // report error in case of invalid json
347:         if (null == $result && null != $response) {
348:             $result = new NWRequestAnswer('Response is not valid JSON.', self::STATUS_CODE_INVALID_JSON);
349:             $result->response = (object) ['success' => false, 'data' => $response, 'responseType' => 'EXCEPTION', 'message' => null, 'title' => null];
350:         }
351:         // if this is not a login request or auto-login is disabled check for problems with session
352:         if ($autoLogin && 'login' != $apiMajor && (null != $apiMajor || 'login' != $this->config['APIVersionMajor'])) {
353:             if (self::STATUS_CODE_EXPIRED == $result->statusCode || self::STATUS_CODE_AUTH_ERR == $result->statusCode) {
354:                 $login = $this->login();
355:                 if (self::STATUS_CODE_OK == $login->statusCode) {
356:                     return $this->request($method, $resource, $content, $apiMajor, $apiMinor, $encoding, false);
357:                 }
358:             }
359:         } 
360:         return $result;
361:     }

363:     /**
364:      * Startet eine Session.
365:      * @param string $username Optional. Der Benutzername, der zum Login verwendet
366:      * werden soll. Wird kein Benutzername übergeben, wird der im Konstruktor angegebene
367:      * Nutzername verwendet.
368:      * @param string $password Optional. Das Passwort, das zum Login verwendet werden
369:      * soll. Wird kein Passwort übergeben, wird das im Konstruktor angegebene Passwort
370:      * verwendet.
371:      * @return object Die Antwort auf die Login-Anfrage.
372:      */
373:     public function login($username = null, $password = null) {
374:         if (null == $username) {
375:             $username = $this->config['username'];
376:         }
377:         if (null == $password) {
378:             $password = $this->config['password'];
379:         }
380:         $content = ['Login' => $this->config['serverLoginLoginValue'], 'username' => $username, 'password' => $password];
381:         $response = $this->request(self::METHOD_POST, 'nami/auth/manual/sessionStartup', $content, 'login', null, self::ENCODING_URL);
382:         $this->apiSessionName = $response->apiSessionName;
383:         $this->apiSessionToken = $response->apiSessionToken;
384:         $this->lastLoginResponse = $response;
385:         if ($this->config['saveCookieInFile']) {
386:             $pathParts = pathinfo($this->config['cookieFile']);
387:             // create folder if it does not exist
388:             if (!file_exists($pathParts['dirname'])) {
389:                 mkdir($pathParts['dirname'], intval($this->config['createDirectoryMode'], 0), true);
390:             }
391:             $file = @fopen($this->config['cookieFile'], 'w');
392:             if ($file) {
393:                 $success = fwrite($file, $this->apiSessionName . '=' . $this->apiSessionToken);
394:             }
395:             if (!($file && $success)) {
396:                 throw new RuntimeException('Could not write to file '
397:                         . var_export($this->config['cookieFile'], true) . '.');
398:             }
399:         }
400:         return $response;
401:     }
